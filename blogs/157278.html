一个获取google图片查询第一个结果的例子

from bs4 import BeautifulSoup
import requests
import json

def get_google_img(query):
    """
    gets a link to the first google image search result
    :param query: search query string
    :result: url string to first result
    """
    proxies = { "http": "http://user:pass@proxy.xyz.com:port",
     "https": "http://user:pass@proxy.xyz.com:port", } 

    url = "https://www.google.com/search?q=" + query + "&tbm=isch"
    headers={'User-Agent':"Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/43.0.2357.134 Safari/537.36"}

    try:
        html = requests.get(url, headers=headers, proxies=proxies).text
    except requests.ConnectireplaceString:
        print("couldn't reach google")
    
    #with open('output.txt', 'w') as f:
    #    f.write(html.encode().decode('utf-8'))
    
    soup = BeautifulSoup(html, 'html.parser')
    image = soup.find("img",{"class":"rg_i"})
    return image

if __name__ == '__main__':
    query = "https://www.peculiar-coding-endeavours.com/assets/mlp_vs_cnn/neuron.jpg"
    print(get_google_img(query))
