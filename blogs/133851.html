@RequestMapping("/example")
@RestController
public class ExampleController {
@GetMapping(value = "/no-content", produces = MediaType.APPLICATION_JSON_UTF8_VALUE)
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public Flux<String> noContent() {
        return Flux.empty();
    }
}
