启动调试设置参考：

{
	"name": "case",
	"type": "cppdbg",
	"request": "launch",
	"program": "/xxx/aaa",
	"args": ["--gtest_catch_exceptions=1", "--gtest_filter=aaaa.xxx"],
	"stopAtEntry": false,
	"cwd": "/xxx",
	"environment": 
	[
	    { 
	     "name": "LD_LIBRARY_PATH",
	     "value": "/xxx" 
	    }    
	],
	"externalConsole": false,
	"MIMode": "gdb",
	"pipeTransport": {
	    "pipeCwd": "", "pipeProgram": "ssh.exe", "pipeArgs": ["root@xx.xx.xx.xx"],
	    "debuggerPath": "export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/xxx;/usr/bin/gdb"
	},
	"sourceFileMap": { "/xxx": "d:/xxx" },
	"setupCommands": [{ "description": "Enable pretty-printing for gdb", "text": "-enable-pretty-printing", "ignoreFailures": true }],
	"logging": { "engineLogging": true }//这个选项会开启调试过程中gdb的打印
}
