创建一个screen session并激活虚拟环境

#!/bin/bash
screen -dmS spawner
screen -S spawner -X stuff 'source 。。。/bin/activate;
'
sleep 3  # Need time for venv screen to initialize.
screen -S spawner -X stuff 'cd 。。。;
'
screen -S spawner -X stuff 'python  。。。/manage.py runserver 0.0.0.0:xxx

相应的离开虚拟环境并终止该session

#!/bin/bash

screen -S spawner -X stuff 'deactivate
'
sleep 2
screen -S spawner -X quit
