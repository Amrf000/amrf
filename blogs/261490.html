例如对于secure-shell-app扩展，它的扩展manifest.json指定的这些文件可以从浏览器侧访问，那么可以通过下面的函数对是否安装或者启用了插件进行检测

function detectExtension(extensionId, callback) {
      window
        .fetch('chrome-extension://' + extensionId + '/html/nassh_google_relay.html')
        .then((res) => {
          const result = true
          callback(result)
        })
        .catch((err) => {
          console.log(err)
          const result = false
          callback(result)
        })
}
