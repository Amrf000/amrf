2.tar -xjvf mwget_0.1.0.orig.tar.bz2

3.cd mwget_0.1.0.orig

4../configure

如果No package 'openssl' found,sudo apt-get install libssl-dev

如果intltool is too old，sudo apt-get install intltool

5.make

6.sudo  make install
