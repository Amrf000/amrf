jsplumb简单使用参考

import { jsPlumb } from 'jsplumb'
//  连接点定义
const dot = {
  isSource: false,
  isTarget: false,
  maxConnections: -1,
  endpoint: [
    'Dot',
    {
      radius: 3
    }
  ]
}
// 获得jsPlumb实例
const plumbIns = jsPlumb.getInstance({
  Container: '外层容器的id',
  PaintStyle: {
    stroke: '#B9C1D4',
    strokeWidth: 1
  }
})
// 创建连线
plumbIns.ready(function() {
  const info = [
    ['Source的id', 'target的id']
  ]
  for (const item of info) {
    plumbIns.connect(
      {
        source: item[0],
        target: item[1],
        anchor: ['Right', 'Left'],
        connector: [
          'Flowchart',
          {
            curviness: 50,
            cornerRadius: 5,
            gap: 3
          }
        ]
      },
      dot
    )
  }
})
