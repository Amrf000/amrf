默认导入导出不带合并单元格信息,可以对上面的例子做一个修改以导出合并信息:

changeMergeCellsToSheet(sheet, mergeCells) {
  let merges = [];
  mergeCells.forEach(item => {
    let tmp = XLSX.utils.decode_range(item);
    merges.push(tmp);
  });
  return merges;
},
var ws = XLSX.utils.aoa_to_sheet(aoa);
